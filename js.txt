document.addEventListener("DOMContentLoaded", function(){
       const square = document.querySelectorAll(".cell");
       const resetButton = document.getElementById("reset");
       const jogadorAtual = "X";
       let gameOver = false;


       square.forEach(function(square) {
        square.addEventListener("click", function(){
          if(!gameOver && square.textContent === ""){
           square.textContent = jogadorAtual;
           square.classList.add(jogadorAtual);

           if(checkWin(jogadorAtual)){
            gameOver = true;
            alert("O jogador " + jogadorAtual + " Ã© o vencedor");
            } else if (checkTie()) {
              gameOver = true;
              alert("Empate!");
            } else {
               jogadorAtual = jogadorAtual === "X" ? "O" : "X";
            }
          }
        });
       });

resetButton.addEventListener("click", function() {
   square.forEach(function(square){
    square.textContent = "";
    square.classList.remove("X", "O");
});

currentPlayer = "X";
gameOver = false;

});

    [0, 4, 8], [2, 4, 6]
    ];

 return winningCombination.some(function(combination) {
  return combination.every(function(index) {

}
